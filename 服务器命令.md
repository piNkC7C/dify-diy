<!-- 环境配置 -->
```shell
# 安装nvm
curl https://raw.githubusercontent.com/creationix/nvm/v0.33.8/install.sh | bash

# 重启配置文件
source ~/.bashrc

# 安装node
nvm install v22.17.0

# 安装pnpm
npm install -g pnpm

# 安装pm2
npm install -g pm2

# 重启配置文件
source ~/.bashrc
```

<!-- 启动项目 -->
```shell
# 第一次启动
# 后端启动,进入到后端目录下
pm2 start "uv run flask run --host 0.0.0.0 --port=5001 --debug" --name dify-api

# 启动异步处理
pm2 start "uv run celery -A app.celery worker -P gevent -c 1 --loglevel INFO" --name dify-worker

# 前端启动,进入到前端目录下
pm2 start "pnpm start" --name dify-web

# 后续重启，pm2给每个进程都标注了id
pm2 restart (id)

# 查看所有进程
pm2 list

# 查看单个进程详细信息
pm2 show (id)
```
